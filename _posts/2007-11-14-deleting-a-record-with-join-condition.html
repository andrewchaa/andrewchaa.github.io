DELETE dbo.Candy  
  FROM dbo.Candy INNER JOIN dbo.CandyReport  
    ON Candy.CandyReportId = CandyReport.CandyReportId  
 WHERE CandyReport.CandyId = @CandyId
 
DELETE dbo.CandyReport
 WHERE CandyReport.CandyId = @CandyId
